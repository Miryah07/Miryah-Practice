
public class Ch5 {

	public static void main(String[] args) {

		int userInput = 120;
		int base = userInput;
		boolean keepGoing = true;

		while(keepGoing){
			for(int testCase = 2; 
					testCase <= base; 
					testCase = testCase + 1){

				if(base%testCase == 0){
					System.out.print(base + ", ");
					base = base / testCase;
					if(base == testCase);{
						break;					
					}
				}
			}
		}
	}
}
