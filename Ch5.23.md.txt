
public class Ch5 {

	public static void main(String[] args) {

		double sumlittleChick = 0;
		double sumbigChick=0;
		boolean keepGoing = true;
		int limit = 1;

		while(keepGoing) {
			for(int i = 1; i <= limit; i++){
				sumlittleChick = sumlittleChick + 1/(double)i;

			}
			for (int i = 1; i >= 1; i--) {
				sumbigChick = sumbigChick + 1/(double)i;

				if(sumbigChick != sumlittleChick){
					keepGoing = false;
					System.out.println(sumlittleChick);
					System.out.println(sumbigChick);
				} else {
					System.out.println("No match for " + limit);
					limit = limit + 1;
					
					System.out.println();

				}
			}


		}

	}
}
